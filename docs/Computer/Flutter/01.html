# List

List students = ["A","B","C"];

students.add("D"); // 向 students 后添加 "D"
students.addAll(["E","F"]); // 向 students 后添加 "E" , "F"
students.remove("A"); // 删除 students 中第一个(索引值最小)匹配内容的一项
students.removeLast(); // 删除 students 中最后一个
students.removeRange(1,3)// 删除索引范围内的元素
    
students.forEach((item){
    //操作逻辑，item是遍历出来的元素
    print(item);
});// 对列表中的每个元素进行操作

//返回 labmda表达式中return的值
students.every((item){
    // 是否都满足条件
    return true;
});

// 返回 Lambda表达式 中满足return true的列表中元素的可迭代对象
students.where((item){
    // 筛选出，满足条件的数据
    return true;
});

// 属性
students.length;
students.last;
students.first;
students.isEmpty;

# 私有与共有

以下划线 _ 命名开头都是私有，否则都是共有 

class Person{
	String? _name = ""; //私有
	int? age = 0;       //共有
    String? _sex = "男";//私有
    Person({this.age,this.name,this.sex});
    Person.createPerson({this.age,this.name,this.sex});
    
    // 私有

        print("$name学习ing");
    }
}

# 异步

Future<String> fetchUserData() {
  return Future.delayed(Duration(seconds: 2), () {
    return "用户数据：张三";
  });
}

void main() {
  print("开始请求...");
  
  // 使用 .then 链式调用
  fetchUserData().then((value) {
    print(value);
  }).catchError((e) {
    print("发生错误: $e");
  }).whenComplete(() => print("请求结束"));

  print("我不会被阻塞，我会先打印");
}

Future<void> displayData() async {
  try {
    print("正在加载...");
    // 等待异步操作结果
    String data = await fetchUserData(); 
    print("展示: $data");
  } catch (e) {
    print("捕获到异常: $e");
  } finally {
    print("加载状态关闭");
  }
}
