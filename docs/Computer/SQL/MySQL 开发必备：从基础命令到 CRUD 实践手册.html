# MySQL 开发必备：从基础命令到 CRUD 实践手册

SQL语句(关键字)不区分大小写

# MySQL登入客户端操作

连接MySQL

mysql -u 用户名 -p

显示当前时间

SELECT now();

退出

exit/quit/ctrl+d

# MySQL数据库操作

查看所有数据库

SHOW DATABASES;

创建数据库

CREATE DATABASE 数据库名字 CHARSET=utf8mb4;

使用数据库

USE 数据库名字;

查看当前使用的数据库

SELECT database();

删除数据库

DROP DATABASE 数据库名字;

# MySQL表操作

查看当前库中的所有表

SHOW TABLES;

创建表

CREATE TABLE 表名(字段名称 数据类型 可选的约束条件);

例子

CREATE TABLE Users (
    user_id INT PRIMARY KEY,          -- 主键，唯一标识每一行
    username VARCHAR(50) NOT NULL,    -- 字符串类型，不能为空
    email VARCHAR(100) UNIQUE,        -- 电子邮件必须唯一
    created_at TIMESTAMP DEFAULT NOW() -- 默认值为当前时间
);
CREATE TABLE Products (
    product_id INT AUTO_INCREMENT,    -- 自动增长 ID，省去手动输入的麻烦
    product_name VARCHAR(100) NOT NULL,
    price DECIMAL(10, 2) CHECK (price > 0), -- 检查约束：价格必须大于 0
    stock_quantity INT DEFAULT 0      -- 如果不输入库存，默认为 0
    PRIMARY KEY (product_id)          -- 另一种定义主键的方式
);
CREATE TABLE Orders (
    order_id INT PRIMARY KEY,
    order_date DATE,
    user_id INT,                      -- 关联 Users 表的 ID
    total_amount DECIMAL(10, 2),
    FOREIGN KEY (user_id) REFERENCES Users(user_id) -- 设置外键约束
);

修改表字段类型

ALTER TABLE 表名 MODIFY 列名 新类型 约束;

删除表

DROP TABLE 表名;

查看表结构

DESC 表名;
-- 或者使用
SHOW CREATE TABLE 表名; -- 查看建表时的完整 SQL

# MySQL CRUD操作

查询数据

-- 查询所有列
SELECT * FROM 表名;
SELECT * FROM 表名 \G; 
-- 查询指定列
SELECT 列1,列2,.. FROM 表名; -- \G 可以把行转列显示，数据多时好排版
-- 例如
SELECT id,name FROM students

插入数据

-- 全列插入：值的顺序必须和字段的顺序一致
INSERT INTO 表名 VALUES(...);
-- 部分列插入 : 值的顺序和给出的列的顺序对应
INSERT INTO 表名(列1..) VALUES(值1...);

修改数据

UPDATE 表名 SET 列1=值1,列2=值2... WHERE 条件;
-- 例如
UPDATE students SET age=18,gender='女' WHERE id=6;

删除数据

delete FROM 表名 WHERE 条件;
